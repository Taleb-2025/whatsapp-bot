// force redeploy - August 17 require('dotenv').config(); const { default: makeWASocket, DisconnectReason, initInMemoryAuthState } = require('@whiskeysockets/baileys'); const P = require('pino'); const qrcode = require('qrcode-terminal'); let sock; // Schritt 1 – Lade auth_info aus Umgebungsvariablen const credsJson = process.env.CREDS_JSON; const keysJson = process.env.KEYS_JSON; if (!credsJson || !keysJson) { console.error("❌ Bitte setze CREDS_JSON und KEYS_JSON als Umgebungsvariablen."); process.exit(1); } const creds = JSON.parse(credsJson); const keys = JSON.parse(keysJson); const { state, saveCreds } = initInMemoryAuthState(); Object.assign(state.creds, creds); Object.assign(state.keys, keys); // Schritt 2 – Bot starten async function startBot() { sock = makeWASocket({ logger: P({ level: 'silent' }), auth: state, printQRInTerminal: true, browser: ['DigiNetz', 'WebApp', '1.0'] }); sock.ev.on('connection.update', async ({ connection, lastDisconnect, qr }) => { if (qr) { console.log("📷 Bitte QR-Code scannen:"); qrcode.generate(qr, { small: true }); } if (connection === 'close') { const shouldReconnect = lastDisconnect?.error?.output?.statusCode !== DisconnectReason.loggedOut; if (shouldReconnect) startBot(); } if (connection === 'open') { console.log('✅ Bot ist verbunden mit WhatsApp'); } }); sock.ev.on('creds.update', async newCreds => { await saveCreds(); console.log("✅ Neue Auth-Daten gespeichert (nur im RAM, für Railway)."); }); sock.ev.on('messages.upsert', async ({ messages }) => { const msg = messages[0]; if (!msg.message || msg.key.fromMe) return; const sender = msg.key.remoteJid; const text = msg.message.conversation || msg.message.extendedTextMessage?.text; if (!text) return; const input = text.trim().toLowerCase(); console.log(`📩 Nachricht von ${sender}: ${text}`); // Startnachricht if (input === 'start' || input === 'hallo diginetz') { await sock.sendMessage(sender, { text: `👋 Willkommen bei *DigiNetz*! Unsere Plattform bietet einfache und schnelle digitale Vorlagen ✅ Bitte wähle eine Option: 1️⃣ Rechnung Kleingewerbe 2️⃣ Rechnung mit MwSt 3️⃣ Privat-Ausgabe 4️⃣ Kontakt 💡 *Tipp*: Speicher diesen Kontakt als *DigiNetz*, um alle Dienste einfacher zu nutzen.` }); return; } // Optionen switch (input) { case '1': await sock.sendMessage(sender, { text: `🧾 *Kleingewerbe Rechnung* Erstelle deine Rechnung hier: 👉 https://diginetz-template.com/kleingewerbe` }); break; case '2': await sock.sendMessage(sender, { text: `📄 *Rechnung mit MwSt (19%)* Jetzt starten: 👉 https://diginetz-template.com/unternehmen` }); break; case '3': await sock.sendMessage(sender, { text: `👤 *Privat-Ausgabe Formular* Deine privaten Ausgaben erfassen: 👉 https://diginetz-template.com/privat-ausgabe` }); break; case '4': await sock.sendMessage(sender, { text: `📞 *Kontakt & Hilfe* Schreib uns direkt: 👉 https://wa.me/491234567890` }); break; default: await sock.sendMessage(sender, { text: `❗ *Ungültige Auswahl*. Bitte 1–4 tippen oder einen Button verwenden.` }); } }); } startBot();
